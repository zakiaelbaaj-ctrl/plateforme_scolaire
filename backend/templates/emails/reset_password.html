# reset_email.py import os import sys import logging from dotenv import load_dotenv import smtplib from email.message import EmailMessage from jinja2 import Environment, FileSystemLoader, select_autoescape import jwt import datetime from urllib.parse import quote_plus # Charger .env load_dotenv() # Logger simple logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s') logger = logging.getLogger(__name__) # Variables requises REQUIRED_VARS = [ 'EMAIL_HOST', 'EMAIL_PORT', 'EMAIL_USER', 'EMAIL_PASS', 'DESTINATAIRE', 'RESET_LINK', 'USER_NAME', 'JWT_SECRET', 'TEMPLATE_DIR', 'TEMPLATE_NAME' ] missing = [v for v in REQUIRED_VARS if not os.getenv(v)] if missing: logger.error("Variables d'environnement manquantes: %s", ", ".join(missing)) sys.exit(1) # Récupération et conversion EMAIL_HOST = os.getenv('EMAIL_HOST') try: EMAIL_PORT = int(os.getenv('EMAIL_PORT')) except ValueError: logger.error("EMAIL_PORT doit être un entier.") sys.exit(1) EMAIL_USER = os.getenv('EMAIL_USER') EMAIL_PASS = os.getenv('EMAIL_PASS') DESTINATAIRE = os.getenv('DESTINATAIRE') RESET_LINK = os.getenv('RESET_LINK').rstrip('/') # enlever slash final si présent USER_NAME = os.getenv('USER_NAME') JWT_SECRET = os.getenv('JWT_SECRET') TEMPLATE_DIR = os.getenv('TEMPLATE_DIR') # ex: 'backend/emails/templates/backend' TEMPLATE_NAME = os.getenv('TEMPLATE_NAME') # ex: 'reset_password.html' # Générer token JWT (expiration 1 heure) payload = { 'user': USER_NAME, 'iat': datetime.datetime.utcnow(), 'exp': datetime.datetime.utcnow() + datetime.timedelta(hours=1) } try: token = jwt.encode(payload, JWT_SECRET, algorithm='HS256') # PyJWT v2 retourne une str ; si bytes, décoder if isinstance(token, bytes): token = token.decode('utf-8') except Exception as e: logger.exception("Erreur lors de la génération du token JWT: %s", e) sys.exit(1) # Construire le lien de réinitialisation en incluant le token (URL-encodé) reset_url = f"{RESET_LINK}?token={quote_plus(token)}" # Configurer Jinja2 avec autoescape pour HTML env = Environment( loader=FileSystemLoader(TEMPLATE_DIR), autoescape=select_autoescape(['html', 'xml']) ) try: template = env.get_template(TEMPLATE_NAME) except Exception as e: logger.exception("Impossible de charger le template Jinja2: %s", e) sys.exit(1) context = { 'reset_link': reset_url, 'user_name': USER_NAME, # autres variables utiles pour le template } html_content = template.render(context) # Préparer EmailMessage msg = EmailMessage() msg['Subject'] = "Réinitialisation de votre mot de passe" msg['From'] = EMAIL_USER msg['To'] = DESTINATAIRE msg.set_content("Pour voir ce message, activez l'affichage HTML.") # fallback texte msg.add_alternative(html_content, subtype='html') # Envoi SMTP: choisir SSL direct (465) ou STARTTLS (autres ports comme 587) use_ssl = EMAIL_PORT == 465 try: if use_ssl: logger.info("Connexion SMTP via SSL sur %s:%d", EMAIL_HOST, EMAIL_PORT) with smtplib.SMTP_SSL(EMAIL_HOST, EMAIL_PORT) as server: server.login(EMAIL_USER, EMAIL_PASS) server.send_message(msg) else: logger.info("Connexion SMTP via STARTTLS sur %s:%d", EMAIL_HOST, EMAIL_PORT) with smtplib.SMTP(EMAIL_HOST, EMAIL_PORT) as server: server.ehlo() server.starttls() server.ehlo() server.login(EMAIL_USER, EMAIL_PASS) server.send_message(msg) logger.info("Email envoyé avec succès à %s", DESTINATAIRE) except smtplib.SMTPAuthenticationError as e: logger.exception("Erreur d'authentification SMTP: %s", e) sys.exit(1) except Exception as e: logger.exception("Erreur lors de l'envoi de l'email: %s", e) sys.exit(1)
